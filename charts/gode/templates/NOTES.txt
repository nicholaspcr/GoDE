Get the HTTP gateway URL:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "gode.selectorLabels" . | replace "\n" "," }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.service.http.port }}:{{ .Values.server.httpPort }}

Then visit: http://localhost:{{ .Values.service.http.port }}
